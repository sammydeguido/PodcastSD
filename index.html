        function navigateTo(pageId) {

            // Hide all sections

            document.querySelectorAll('.page-section').forEach(section => {

                section.classList.add('hidden-section');

            });

            

            // Show target section

            const target = document.getElementById(pageId);

            if(target) {

                target.classList.remove('hidden-section');

                

                // Re-trigger fade animation

                target.classList.remove('fade-in');

                void target.offsetWidth; // trigger reflow

                target.classList.add('fade-in');

            }


            // Update Nav State

            document.querySelectorAll('.nav-link').forEach(link => {

                if(link.getAttribute('data-page') === pageId) {

                    link.classList.add('text-orange-500');

                    link.classList.remove('text-gray-400');

                } else {

                    link.classList.remove('text-orange-500');

                    link.classList.add('text-gray-400');

                }

            });


            // Mobile Menu Close logic

            const mobileMenu = document.getElementById('mobile-menu');

            if(!mobileMenu.classList.contains('hidden')) {

                mobileMenu.classList.add('hidden');

                document.getElementById('mobile-menu-btn').innerHTML = '<i data-lucide="menu"></i>';

                lucide.createIcons();

            }


            // Scroll to top

            window.scrollTo(0, 0);

        }


        // Mobile Menu Toggle

        document.getElementById('mobile-menu-btn').addEventListener('click', function() {

            const menu = document.getElementById('mobile-menu');

            const isHidden = menu.classList.contains('hidden');

            

            if (isHidden) {

                menu.classList.remove('hidden');

                this.innerHTML = '<i data-lucide="x"></i>';

            } else {

                menu.classList.add('hidden');

                this.innerHTML = '<i data-lucide="menu"></i>';

            }

            lucide.createIcons();

        });

        function openBooking(title, price) {

            basePrice = price;

            currentAddonsPrice = 0;

            updateTotal();

            

            // Reset state

            document.getElementById('step-addons').classList.remove('hidden');

            document.getElementById('time-slots').classList.add('hidden');

            

            // Remove previous selections and reset indicators

            document.querySelectorAll('.addon-card').forEach(el => {

                el.classList.remove('selected');

                const indicator = el.querySelector('.selection-indicator');

                indicator.classList.remove('bg-orange-500', 'border-transparent');

                indicator.classList.add('border-gray-500', 'text-transparent');

            });

            document.querySelectorAll('.calendar-day').forEach(el => el.classList.remove('selected'));

            document.querySelectorAll('.time-slot').forEach(el => el.classList.remove('selected'));


            // Set Data

            document.getElementById('modal-title').innerText = title;

            document.getElementById('booking-modal').classList.remove('hidden-modal');

            lucide.createIcons(); // Ensure icons load in the modal

        }


        function toggleAddon(element, price) {

            if (element.classList.contains('selected')) {

                element.classList.remove('selected');

                currentAddonsPrice -= price;

                // Visuals

                const indicator = element.querySelector('.selection-indicator');

                indicator.classList.remove('bg-orange-500', 'border-transparent');

                indicator.classList.add('border-gray-500', 'text-transparent');

            } else {

                element.classList.add('selected');

                currentAddonsPrice += price;

                // Visuals

                const indicator = element.querySelector('.selection-indicator');

                indicator.classList.add('bg-orange-500', 'border-transparent');

                indicator.classList.remove('border-gray-500', 'text-transparent');

            }

            updateTotal();

        }


        function updateTotal() {

            const total = basePrice + currentAddonsPrice;

            document.getElementById('modal-price').innerText = '$' + total.toFixed(2);

            document.getElementById('pay-btn-amount').innerText = 'Pay $' + total.toFixed(2);

        }
